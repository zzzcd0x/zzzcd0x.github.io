线段树

Posted on 2020-07-14，14 min read

组成

1号结点为根结点，存储原数组区间1到6的数的和

左儿子下标为2，求法为1>>1,存储[1,3]的和

右儿子下标为3，求法为1>>1|1,存储[4,6]的和

这里3和4时区间中点和中点+1

用二进制运算比较快，处理大量数据时常数可以低一点

其实线段树能维护的东西远不止和，还可以维护区间最大值、最小值、最大子段和等等

怎么维护呢

先从建树开始

从根节点开始递归建树 开四倍空间

1.数组写法

int a[1005];  //原数组
int t[4005]; //线段树

void build(int now,int l,int r){

    if(l == r)
    {
        t[now] = a[l];
        return;
    }

    int mid = (l+r)>>1;

    build(now<<1,l,mid);
    build(now<<1|1,mid+1,r);

    pushup(now); //回溯的时候更新父亲节点
}

2.结构体写法

    int l,r;

}

区间查询

  1. 当前查找到的区间被包含在需要查找的区间内则直接返回
  2. 否则通过中点判断要查找到区间和左右子树哪个有交集
  3. 如果都有则将两边的查找结果进行相应的处理后返回
  4. 思想类似二分查找
int find(int now,int L,int R){

    int l = node[now].l;
    int r = node[now].r;
    if(L <= l && R >= r)
        return node[now].sum;
    
    int mid = (l+r)>>1;

    if(mid >= R)
        return find(now<<1,L,R);
    else if(mid < l)
        return find(now<<1|1,L,R);
    else
    {
        int tot = 0;
        tot += find(now<<1,L,R);
        tot += find(now<<1|1,L,R);
        return tot; 
    }
}

区间修改

大体步骤和区间查询类似

标记下放: 如果每次修改都去相应的修改涉及到的所有的结点的话会产生较大的复杂度，因此考虑使用标记下放的方法，即当该结点被访问到的时候再将标记下放，使子树得到正确的信息。所以区间修改只更新到需要修改的区间的真子集即可，但在此之前每一个访问到的结点都要进行标记下放，不然需要更新的区间得不到正确的更新信息

int tag[40005];

void pushdown(int now){

    if(tag[now])
    {
        tag[now<<1] += tag[now];
        tag[now<<1|1] += tag[now];
        node[now<<1].sum += tag[now]*(node[now<<1].r-node[now<<1]+1);
        node[now<<1|1].sum += tag[now]*(node[now<<1|1].r-node[now<<1|1]+1);
        tag[now] = 0;
    }

}

void modify(int now,int L,int R,int k){

    int l = node[now].l;
    int r = node[now].r;

    if(L <= l && R >= r)
    {
        tag[now] += k;
        node[now].sum += k*(r-l+1);
    }

    pushdown(now);

    int mid = (l+r)>>1;
    if(mid >= L) modify(now<<1,L,R,k);
    if(R > mid) modify(now<<1|1,L,R,k);

    pushup(now);
}
  • pushup的作用是当某一个结点的子树的值被修改了则去用修改后的子树的信息去更新当前结点的信息

维护最大子段和

什么是最大子段和？

其实就是当前线段树区间内，一个任意的连续的序列的和的最大值

其他会用到的概念:

最大左子段和:从区间左端点开始的最大子段和

最大右子段和:从区间右端点开始的最大子段和

pushup

最大左子段和可能是左子树的最大左子段和也可能是左子树的区间总和加右子树的最大左子段和，最大右子段和同理

最大子段和可能是左子树的最大子段和或右子树的最大子段和或左子树的最大右子段和+右子树的最大左子段和

void pushup(int now){

区间查找

Node ask(int now,int L,int R){

    int l = node[now].l;
    int r = ndoe[now].r;

    if(L <= l && R >= r)
        return node[now];
    
    int mid = (l+r)>>1;

    if(mid >= R) return ask(now<<1,L,R);
    else if(mid<L) return ask(now<<1|1,L,R);
    else{
        Node ans;
        Node left = ask(now<<1,L,R);
        Node right = ask(now<<1|1,L,R);
        ans.sum = left.sum + right.sum;
        ans.lm = max(left.lm,left.sum+right.lm);
        ans.rm = max(right.rm,right.sum+left.rm);
        ans.Max = max(left.rm+right.lm,max(left.Max,right.Max));
        return ans;
    }
}

最大子段和

题目链接

#include<cstdio>
#include<algorithm>
using namespace std;

int n,m;
int a[500005];

struct Node{

    int l,r;
    int sum;
    int Max;
    int lm,rm;
}node[2000005];

void pushup(int now){

    node[now].sum = node[now<<1].sum+node[now<<1|1].sum;
    node[now].lm = max(node[now<<1].lm,node[now<<1].sum+node[now<<1|1].lm);
    node[now].rm = max(node[now<<1|1].rm,node[now<<1|1].sum+node[now<<1].rm);
    node[now].Max = max(max(node[now<<1].Max,node[now<<1|1].Max),node[now<<1].rm+node[now<<1|1].lm);
}

void build(int now,int l,int r){

    node[now].l = l;
    node[now].r = r;

    if(l == r)
    {
        node[now].sum = a[l];
        node[now].lm = a[l];
        node[now].rm = a[l];
        node[now].Max = a[l];
        return;
    }
    int mid = (l+r)>>1;

    build(now<<1,l,mid);
    build(now<<1|1,mid+1,r);

    pushup(now);
}

void modify(int now,int to,int k){

    int l = node[now].l;
    int r = node[now].r;

    if(l == r)
    {
        // a[now] = k;
        node[now].sum = k;
        node[now].Max = k;        
        node[now].lm = k;
        node[now].rm = k;

        return;
    }
    int mid = (l+r)>>1;

    if(to <= mid) modify(now<<1,to,k);
    else modify(now<<1|1,to,k);

    pushup(now);
}

Node ask(int now,int L,int R){

    int l = node[now].l;
    int r = node[now].r;

    if(L <= l && R >= r) return node[now];

    int mid = (l+r)>>1;

    if(mid >= R) return ask(now<<1,L,R);
    else if(mid < L) return ask(now<<1|1,L,R);
    else
    {
        Node ans;
        Node left = ask(now<<1,L,R);
        Node right = ask(now<<1|1,L,R);
        ans.sum = left.sum + right.sum;
        ans.lm = max(left.lm,left.sum+right.lm);
        ans.rm = max(right.rm,right.sum+left.rm);
        ans.Max = max(left.rm+right.lm,max(left.Max,right.Max));
        return ans;
    }
}

int main(){

    int k,x,y;

    scanf("%d %d",&n,&m);

    for(int i = 1; i <= n; i++)
        scanf("%d",&a[i]);

    build(1,1,n);
    
    for(int i = 1; i <= m; i++)
    {
        scanf("%d %d %d",&k,&x,&y);
        if(k == 1)
        {
            if(x > y)
            {
                int t = x;
                x = y;
                y = t;
            }
            Node ans = ask(1,x,y);
            printf("%d\n",ans.Max);
        }
        if(k == 2)
            modify(1,x,y);
    }

    return 0;
}

线段树+状压

题目链接

本题是给一个木板的不同区间染色，一共有30种颜色，可以考虑用一个二进制数字来储存该区间内的数被染了哪一种颜色，线段树中第i个结点存储的数字的二进制第j位为1，则该结点被染成了颜色j，若为0则没有染成颜色j，区间颜色个数统计使用按位与运算，将最终结果进行二进制拆分统计为1的个数即可。

#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int MAXN = 1e5+5;

int n,t,o;
int le[MAXN<<2];
int ri[MAXN<<2];
int tag[MAXN<<2];
int color[MAXN<<2];

inline int read() {
    int x = 0;
    char c = getchar();
    while(c < '0' || c > '9')
        c = getchar();
    while(c >= '0' && c <= '9') {
        x = x*10 + c-'0';
        c = getchar();
    }
    return x;
}

int ls(int p) {return p<<1;}
int rs(int p) {return p<<1|1;}
void pushup(int p) {color[p] = color[ls(p)] | color[rs(p)];}

void upd(int p,int v) {
    color[p] = v;
    tag[p] = v;
}

void pushdown(int p) {
    if(tag[p]) {
        upd(ls(p),tag[p]);
        upd(rs(p),tag[p]);
        tag[p] = 0;
    }
}

void build(int now,int l,int r) {
    le[now] = l;
    ri[now] = r;
    color[now] = 0;
    tag[now] = 0;
    if(l == r) {
        color[now] = 1;
        return;
    }
    int mid = (l+r)>>1;
    build(ls(now),l,mid);
    build(rs(now),mid+1,r);
    pushup(now);
}

void modify(int now,int l,int r,int v) {
    int L = le[now];
    int R = ri[now];
    if(r < L || l > R) return;
    if(l <= L && r >= R) {
        tag[now] = 1<<(v-1);
        color[now] = 1<<(v-1);
        return;
    }
    pushdown(now);
    int mid = (L+R)>>1;
    if(l <= mid) modify(ls(now),l,r,v);
    if(r > mid) modify(rs(now),l,r,v);
    pushup(now);
}

int query(int now,int l,int r) {
    int L = le[now];
    int R = ri[now];
    if(r < L || l > R) return 0;
    if(l <= L && r >= R) {
        return color[now];
    }
    int ans = 0;
    int mid = (L+R)>>1;
    pushdown(now);
    if(l <= mid) ans |= query(ls(now),l,r);
    if(r > mid) ans |= query(rs(now),l,r);
    return ans;
}

int main() {
    while(~scanf("%d %d %d",&n,&t,&o)) {
        int x, y, z;
        char c[2];
        build(1,1,n);
        while(o--) {
            scanf("%s",c);
            if(c[0] == 'C') {
                scanf("%d %d %d",&x,&y,&z);
                if(x > y)
                    swap(x,y);
                modify(1,x,y,z);
            }
            else if(c[0] == 'P') {
                scanf("%d %d",&x,&y);
                if(x > y)
                    swap(x,y);
                int cnt = 0;
                int tot = query(1,x,y);
                while(tot) {
                    if(tot&1) cnt++;
                    tot >>= 1;
                }
                printf("%d\n",cnt);
            }
        }
    }

    return 0;
}
