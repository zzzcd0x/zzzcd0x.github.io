    for(int i = 1; i <= n; i++)
    {
        int Min = inf;

            if(!b[j])
                if(Min == inf || d[j]<d[Min])
                    Min = j;
        if(Min == inf) return ;
        b[Min] = true;
        for(int i = head[Min]; i ;i = edge[i].next)
        {
            int to = edge[i].to;
            int val = edge[i].val;
            if(!b[to])
                d[to] = d[Min]+val;
        }
    }
}

#include<queue>
#include<cstdio>
#include<utility>
#include<string.h>
#include<algorithm>
using namespace std;
typedef long long ll;
const int MAXN = 1e3+5;
const int MAXP = 1e4+5;

inline int read() {

    int x = 0;
    int k = 1;
    char c = getchar();
    while(c < '0' || c > '9') {
        if(c == '-')
            k = -1;
        c = getchar();
    }
    while(c >= '0' && c <= '9') {
        x = x*10 + c-'0';
        c = getchar();
    }
    return x;
}

int tot;
int n, p, k;
int dis[MAXN];
bool vis[MAXN];
int head[MAXN];
ll val[MAXP<<1];
int to[MAXP<<1];
int nxt[MAXP<<1];
int flag[MAXP<<1];

void add(int x,int y, ll z) {
    nxt[++tot] = head[x];
    head[x] = tot;
    to[tot] = y;
    val[tot] = z;
}

bool dijkstra(ll mid) {
    memset(dis,0x3f,sizeof(dis));
    memset(vis,false,sizeof(vis));

    priority_queue<pair<int,int> > q;

    dis[1] = 0;
    q.push(make_pair(0,1));

    while(!q.empty()) {
        int x = q.top().second;
        q.pop();
        if(vis[x]) continue;
        vis[x] = true;

        for(int i = head[x]; i ; i = nxt[i]) {
            if(dis[to[i]] > dis[x] + flag[i]) {
                dis[to[i]] = dis[x] + flag[i];
                if(!vis[to[i]])
                    q.push(make_pair(-dis[to[i]],to[i]));
            }
        }
    }

    if(dis[n] > k)
        return false;
    return true;
}

void inp() {

    n = read();
    p = read();
    k = read();

    for(int i = 1; i <= p; i++) {
        int x = read();
        int y = read();
        ll z = read();
        add(x,y,z);
        add(y,x,z);
    }
}

int main() {

    inp();

    int l = 0;
    int r = 1e6;
    ll ans = -1;

    while(l <= r) {
        ll mid = (l+r)>>1;
        
        for(int i = 1; i <= tot; i++) {
            if(val[i] <= mid)
                flag[i] = 0;
            else flag[i] = 1;
        }

        if(dijkstra(mid)) {
            ans = mid;
            r = mid-1;
        }
        else l = mid+1;
    }

    printf("%lld\n",ans);

    return 0;
}

#include<cstdio>
#include<queue>
#include<stack>
#include<utility>
#include<string.h>
#include<algorithm>
using namespace std;
const int MAX = 1e5+5;
typedef long long ll;

int n, m, s, t;

inline int read() {

    int x = 0;
    char c = getchar();
    while(c < '0' || c > '9')
        c = getchar();
    while(c >= '0' && c <= '9') {
        x = x*10 + c-'0';
        c = getchar();
    }

    return x;
}

int tot;
int path[MAX];
int head[MAX];
int to[MAX<<1];
int nxt[MAX<<1];
int val[MAX<<1];
int from[MAX<<1];

void add(int x,int y, ll z) {

    nxt[++tot] = head[x];
    head[x] = tot;
    to[tot] = y;
    val[tot] = z;
    from[tot] = x;
}

void inp() {

    t = n;
    s = 1;
    for(int i = 1; i <= m; i++) {
        int x = read();
        int y = read();
        ll z = (ll) read();
        add(x,y,z);
        add(y,x,z);
    }
}

ll d[MAX];
bool vis[MAX];

void dijkstra() {

    priority_queue<pair<int,int> > q;
    memset(d,0x3f,sizeof(d));
    memset(path,0,sizeof(path));
    memset(vis,false,sizeof(vis));

    d[s] = 0;
    q.push(make_pair(0,s));

    while(!q.empty()) {
        int x = q.top().second;
        q.pop();
        if(vis[x]) continue;
        vis[x] = true;
        for(int i = head[x]; i ; i = nxt[i]) {
            if(d[to[i]] > d[x] + val[i]) {
                d[to[i]] = d[x] + val[i];
                if(!vis[to[i]])
                    q.push(make_pair(-d[to[i]],to[i]));
                path[to[i]] = x; 
            }
        }
    }
}

void oup() {

    int t = n;
    stack<int> temp;
    bool flag = false;

    while(t) {

        temp.push(t);
        t = path[t];
        if(t == 1)
            flag = true;
    }

    if(flag)
        while(!temp.empty()) {
            printf("%d ",temp.top());
            temp.pop();
        }
    else printf("-1");
    puts("");
}

int main() {

    while(~scanf("%d %d",&n,&m)) {

        inp();

        dijkstra();

        oup();
    }

    return 0;
}
