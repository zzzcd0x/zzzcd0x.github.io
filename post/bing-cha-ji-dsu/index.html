    什么是并查集

    并查集是一种比较简洁但功能强大的数据结构，主要用于维护一些不相关集合。

    支持合并两个集合、查找一个结点属于哪一个集合两种操作。

    初始化

    在开始使用并查集之前认为每一个点都代表一个独立的集合

    for(int i = 1; i <= n; i++) 
	dad[i] = i;

    查询操作

    将关联两个结点的关系看作图中的边，则查询操作就是不断查找一个结点的父结点，直至找到根节点，即代表该集合的结点。

    朴素写法

    int anc(int p) {
    if(dad[p] == p) return p;
    else return anc(dad[p]);
}

    这样查找在整个关系构成一条链时时间复杂度会达到 O(n)O(n)O(n)。

    因此诞生了路径压缩优化

    路径压缩

    在递归找到根节点后，回溯的过程中将子节点的父亲结点设为找到的根节点，这样就可以避免路径的重复访问

    int anc(int p) {
    if(dad[p] == p) return p;
    else return dad[p] = anc(p);
}

    这样单次查找的时间复杂度就变成 O(1)O(1)O(1) 了，总时间复杂度均摊 O(logn)O(logn)O(logn) 。

    合并操作

    并查集中用一个数组用于储存每个集合中的代表元素，因此在合并两个集合时只要将这两个集合的代表元素合并到同一个集合中即可。

    void uni(int x,int y) {
    x = anc(x);
    y = anc(y);
    if(x != y)
        dad[x] = y;
}

    种类并查集

    在某些情况下会出现类似敌人的敌人是朋友的状态需要维护，这种情况下就需要使用种类并查集进行维护

    种类并查集需要开两倍大小的空间用于维护状态（有的时候不止两倍），其中 1−n1-n1−n 用于维护朋友关系，n+1−2nn+1-2nn+1−2n 用于维护敌人关系。

    这样在合并操作时，同一个结点的敌人就会被合并到同一个集合里。

    与前面的并查集不同的是，假设要设定 xxx 和 yyy 为敌人关系，则合并 xxx 与 y+ny+ny+n 和 x+nx+nx+n 与 yyy 所在集合。

    并查集裸题，直接打模板即可

    #include<cstdio>
using namespace std;

int n, m, p;
int dad[5005];

int anc(int x){
	if(dad[x] == x) return x;
    return dad[x] = anc(dad[x]);
}

void uni(int x, int y){

    x = anc(x);
    y = anc(y);
    
    if(x != y)
        dad[x] = y;
}

bool ask(int x,int y)
{
    return anc(x) == anc(y);
}

void inp(){

    int x, y;
    scanf("%d %d %d",&n,&m,&p);
    for(int i = 1; i <= m; i++)
    {
        scanf("%d%d",&x,&y);
        uni(x,y);
    }
}

void getans(){

    int x, y;
    for(int i = 1; i <= p; i++)
    {
        scanf("%d %d",&x,&y);
        if(ask(x,y))
            printf("Yes\n");
        else
            printf("No\n");
    }
}

int main()
{
    inp();

    getans();

    return 0;
}

    题目链接

    本题要求冲突值尽可能小，则先对所有关系进行从大到小排序，优先合并冲突值大的关系，则第一次出现冲突是即为冲突值的最小值

    #include<cstdio>
#include<algorithm>
using namespace std;
const int MAXN = 2e4+5;
const int MAXM = 1e5+5;

struct Node{
    int x;
    int y;
    int num;
}node[MAXM];

inline int read() {
    int x = 0;
    char c = getchar();
    while(c < '0' || c > '9')
        c = getchar();
    while(c >= '0' && c <= '9') {
        x = x*10 + c-'0';
        c = getchar();
    }
    return x;
}

int ans;
int n, m;
int dad[MAXN<<1];

int anc(int p) {
    if(dad[p] == p) return p;
    return dad[p] = anc(dad[p]);
}

void uni(int x,int y) { dad[anc(x)] = anc(y);}

bool jud(Node& x,Node& y) {return x.num > y.num;}

int main() {
    n = read();
    m = read();

    for(int i = 1; i <= n; i++) {
        dad[i] = i;
        dad[i+n] = i+n;
    }

    for(int i = 0; i < m; i++) {
        node[i].x = read();
        node[i].y = read();
        node[i].num = read();
    }
    sort(node,node+m,jud);

    for(int i = 0; i < m; i++) {
        if(anc(node[i].x) == anc(node[i].y)) {
            ans = node[i].num;
            break;
        }
        uni(node[i].x,node[i].y+n);
        uni(node[i].x+n,node[i].y);
    }

    printf("%d\n",ans);

    return 0;
}
