    最大正方形

    题目链接

    题目描述

    给定一个 n∗mn*mn∗m 的只包含0和1的矩阵，求出矩阵中不包含0的最大正方形

    解题思路

    求出给定矩阵的二维前缀和，枚举每一个可能的顶点和以该顶点做左上角的每一个可能的正方形，差分求出该矩阵中1的个数，如果1的个数等于正方形面积，则为一个合法矩形。

    AC代码

    #include<cstdio>
#include<ctype.h>
#include<algorithm>
using namespace std;
const int MAXN = 1e2+5;

template<class T> inline bool read(T &x) {
    x = 0;
    bool f = false;
    char c = getchar();
    while(!isdigit(c)) {
        if(c == EOF) return false;
        f ^= c == '-';
        c = getchar();
    }
    while(isdigit(c)) {
        x = (x<<3) + (x<<1) + (c^48);
        c = getchar();
    }
    if(f) x = -x;
    return true;
}

int l;
int n, m;
int sum[MAXN][MAXN];
int Map[MAXN][MAXN];

void inp() {
    read(n);
    read(m);
    l = min(n,m);
    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= m; j++) {
            read(Map[i][j]);
            sum[i][j] = sum[i-1][j] + sum[i][j-1] - sum[i-1][j-1] + Map[i][j];
        }
}

int main() {
    inp();
    int ans = 1;
    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= m; j++) {
            for(int k = 1; k <= l && i-k >= 0 && j-k >= 0; k++) {
                int tox = i-k;
                int toy = j-k;
                int x = sum[tox][toy] - sum[i][toy] - sum[tox][j] + sum[i][j];
                if(x == k*k && k > ans)
                    ans = k;
            }
        }
    printf("%d\n",ans);
    return 0;
}

    Tallest Cow S

    题目链接

    题目描述

    FJFJFJ 有 nnn 头奶牛，他们按顺序排成一排，FJFJFJ 只知道这些奶牛中最高的奶牛是第几个，以及这头奶牛的身高，现在 FJFJFJ 有 RRR 条信息，每条信息有两头奶牛的编号 aaa 和 bbb ，表示 bbb 的身高一定大于等于 aaa 的身高，且 aaa 和 bbb 之间的奶牛身高一定低于 aaa 和 bbb ，FJFJFJ 想根据这些信息知道每一头奶牛可能的最大身高。

    解题思路

    对于 RRR 条信息，将每一条信息中给出的 aaa 和 bbb 之间的奶牛身高减一，最终根据最高的奶牛的身高还原出序列即可。注意信息判重

    AC代码

    #include<cstdio>
#include<ctype.h>
#include<algorithm>
using namespace std;
const int MAXN = 1e4+5;

template<class T>inline bool read(T &x) {
    x = 0;
    register bool f = false;
    register char c = getchar();
    while(!isdigit(c)) {
        if(c == EOF) return false;
        f ^= c == '-';
        c = getchar();
    }
    while(isdigit(c)) {
        x = (x<<3) + (x<<1) + (c^48);
        c = getchar();
    }
    if(f) x = -x;
    return true;
}
template<class T> inline void print(T x) {
    if(x < 0) {
        putchar('-');
        x = -x;
    }
    if(x > 9) print(x/10);
    putchar(x%10^48);
}
template<class T,class ...S> inline bool read(T &x,S &...y) {return read(x) && read(y...);}
template<class T,class ...S> inline void print(T x,char c) {print(x),putchar(c);}

int n,r,I,h;
int f[MAXN];
int num[MAXN];
bool vis[MAXN][MAXN];

int main() {
    read(n,I,h,r);
    num[I] = h;

    for(register int i = 1; i <= r; i++) {
        int x, y;
        read(x,y);
        if(x > y) swap(x,y);
        if(vis[x][y]) continue;
        vis[x][y] = true;
        f[x+1]--;
        f[y]++;
    }
    for(register int i = I+1; i <= n; i++) {
        num[i] = num[i-1] + f[i];
    }
    for(register int i = I; i >= 1; i--) {
        num[i-1] = num[i] - f[i];
    }
    for(register int i = 1; i <= n; i++)
        print(num[i],'\n');

    return 0;
}

    IncDec Sequence

    题目链接

    题目描述

    给定一个长度为 nnn 的数列 a1,a2,⋯ ,an{a_1,a_2,\cdots,a_n}a1​,a2​,⋯,an​ ，每次可以选择一个区间 [l,r][l,r][l,r] ，使这个区间内的数都加 1 或者都减 1。

    请问至少需要多少次操作才能使数列中的所有数都一样，并求出在保证最少次数的前提下，最终得到的数列有多少种。

    解题思路

    用差分考虑这道题，数列中所有的数都一样即差分数组中除第一个数外，其他数都为0。操作时可以选一个数加一或者减一，这样的话差分数组中的正数和负数就可以互相消，如果找不到可以一起消的数就和 1 或 nnn 消，这样不影响差分数组中其他数的值。可以看出，最少的操作次数一定是原始差分数组中最大的正数或最小的负数。

    对于第二个子任务，消数时的目标是将其他数消到等于数列的第一个数，因此第一个数可能的数即为答案，当整个差分数组中只剩下正数或负数时，这些数只能和第一个数消，也可以选择跟第 nnn 个数消，答案也就是这些数跟第一个数消的时候产生的。所以答案为差分数组中最大的正数与最小的负数的绝对值的差加一。

int n;
ll a,b;
ll x[MAXN];
ll num[MAXN];

int main() {
    read(n);
    for(register int i = 1; i <= n; i++) {
        read(num[i]);
        x[i] = num[i]-num[i-1];
        if(i >= 2) {
            if(x[i] > 0) a += x[i];
            else b += x[i];
        }
    }
    b = -b;
    print(max(a,b),'\n');
    print(1+abs(a-b));
    return 0;
}
