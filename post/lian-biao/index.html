链表

2020-07-18

什么是链表

  • 链表是一种物理存储单元上非连续、非顺序的存储结构，数据元素的逻辑顺序通过链表中的指针链接次序实现
  • 链表最大的特点是不以下标来寻找某一元素，而是通过相邻元素中的联系来寻找，也可以理解为链表不需要每一个元素相邻，只需要他们记住自己旁边的元素
  • 链表又分为单向链表、双向链表、循环链表
  • 一个简单的链表仅需要支持插入、删除、查询三种操作

单向链表

  • 单向链表中的每一个元素记住自己前一个或后一个元素的地址，因此在查找元素时只能单向查找，即没有回头的机会
    下面我们来创建一个顺序为1，2，4，3的单向链表

  • 插入结点

    假设我们要将5插入到2的后面

  1. 首先从头结点开始找到储存2的结点
  2. 然后将需要插入的结点的指针指向2的下一个结点，也就是4
  3. 最后将2的指针指向5
    也就是这样的一个过程

  • 删除结点

    假设我们不喜欢刚刚插入的5想要把他删除

  1. 首先我们要找到下一个结点是5的结点所在的位置
  2. 然后将这个结点的指针指向5结点的指向的下一个结点的位置
  3. 最后将储存5结点的指针指向NULL

  • 查询结点

    如果我们要查询某个结点的位置，我们只需要从头结点开始从搜索，直到找到该结点

  • 操作实现

    下面我们使用指针来建立一个单向链表实现上述操作

#include<iostream>
using namespace std;

struct point{

    int key;
    point* next;

    point(int _key, point* _next)
    {
        next = _next;
        key = _key;
    }

};

point *root;

void init(){           //初始化链表

    root = new point(-1,NULL);    //在链表最开始的位置扔一个伪结点
}

void ins(int x, int y){           //在x后面插入y

    for(point*p = root; p != NULL; p = p->next)
    {
        if(p->key == x)          //找到存储x的结点
        {
            point *n = new point(y,p->next);  //新建存储y的结点
            p->next = n;         //调整存储x的结点的指针
            break;
        }
    }
}

void ask(int x){                 //查询是否有存储x的结点

    for(point*p = root->next; p != NULL; p=p->next)
        if(p->key == x)
            cout << p->next << endl;
}

void del(int x){                 //删除存储x的结点

    for(point*p = root; p ; p=p->next)
        if(p->next->key == x)   //找到存储x的结点的前一个结点
        {
            point *n = p->next;
            p->next = p->next->next;
            delete n;
            break;
        }

}

void out(){                    //输出链表

    for(point*p = root->next; p ; p=p->next)
    {
        cout << p->key << " ";
    }
    cout << endl;
}

int main(){

    init();

    ins(-1,1);
    ins(1,2);
    ins(2,4);
    ins(4,3);
    out();

    ins(2,5);
    out();

    del(5);
    out();

    ask(4);
    out();

    return 0;
}

  • 双向链表

  • 双向链表与单向链表不同的地方就是双向链表的每一个结点存储了他前一个结点和后一个结点的位置
    用双向链表来实现上述的三个步骤的过程为

  • 插入结点

  • 删除结点

  • 查询结点

  • 实现

  • 指针写法

    对单向链表进行部分修改即可得到双向链表

#include<iostream>
using namespace std;

struct point{

    int key;
    point* next;
    point* front;

    point(int _key, point* _next, point* _front)
    {
        next = _next;
        key = _key;
        front = _front;
    }

};

point *root;

void init(){           //初始化链表

    root = new point(-1,NULL,NULL);    //在链表最开始的位置扔一个伪结点
}

void ins(int x, int y){           //在x后面插入y

    for(point*p = root; p != NULL; p = p->next)
    {
        if(p->key == x)          //找到存储x的结点
        {
            point *n = new point(y,p->next,p);
            if(p->next)          //判断p是不是尾结点
                p->next->front = n;
            p->next = n;
        }
    }
}

void ask(int x){                 //查询是否有存储x的结点

    for(point*p = root->next; p != NULL; p=p->next)
        if(p->key == x)
            cout << p->next << endl;
}

void del(int x){                 //删除存储x的结点

    for(point*p = root; p ; p=p->next)
        if(p->next->key == x)   //找到存储x的结点的前一个结点
        {
            point *n = p->next;
            p->next = p->next->next;
            delete n;
            break;
        }

}

void out(){                    //输出链表

    for(point*p = root->next; p ; p=p->next)
    {
        cout << p->key << " ";
    }
    cout << endl;
}

int main(){

    init();

    ins(-1,1);
    ins(1,2);
    ins(2,4);
    ins(4,3);
    out();

    ins(2,5);
    out();

    del(5);
    out();

    ask(4);
    out();

    return 0;
}

  • 数组写法

#include<iostream>
using namespace std;

struct point{
    int pre = 0;    //默认每一个结点前驱和后继为0
    int next = 0;
    int key;
};

point p[105];
int root = 0;
int cnt = 0;

void init(){       //初始化

    p[root].key = -1;
    p[root].next = 0;
    p[root].pre = 0;
}

void ins(int x,int y){
    
    int i = root;
    do
    {
        if(p[i].key == x)
        {
            p[++cnt].key = y;
            p[cnt].next = p[i].next;
            p[cnt].pre = i; 
            if(p[i].next)
                p[p[i].next].pre = cnt;
            p[i].next = cnt;
            break;
        }
        i = p[i].next;
    }while(i != 0);
}

void ask(int x){

    int i = root;
    do
    {
        if(p[i].key == x)
        {
            cout << i << endl;
            return;
        }
        i = p[i].next;
    }while(i != 0);
}

void del(int x){

    int le = p[x].pre;
    int rt = p[x].next;
    p[le].next = rt;
    p[rt].pre = le;
}

void out(){

    for(int i = p[root].next; i != 0; i = p[i].next)
        cout << p[i].key << " ";

    cout << endl;
}

int main(){

    init();

    ins(-1,1);
    ins(1,2);
    ins(2,4);
    ins(4,3);
    out();

    ins(2,5);
    out();

    del(5);
    out();

    ask(4);
    out();

    return 0;
}
