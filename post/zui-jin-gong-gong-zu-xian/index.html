最近公共祖先

2020-07-25 图论

树上倍增法

怎么计算的？

假设要计算树上x和y结点的最近公共祖先，d[x]表示结点x的深度，F[x,k]表示结点x向上跳2

设d[x] >= d[y]否则交换x和y

将x结点尝试向上走2^{log(n)}...2^{log(0)}步，检查每次到达的结点是否比y深，若是，则令x=F[x,k]，跳出循环

如果此时x=y则两个结点的LCA就是y

如果不是，则将x和y同时向上调整，保持两者深度相同但不相等，即将两节点同时向上跳2^{log(n)}...2^{log(0)}步，若此时F[x,k]!=F[y,k](即深度不相等)，则令x=F[x,k],y=F[y,k]

循环结束后x和y一定差一步到达同一结点，即他们的最近公共祖先为F[x,0]或F[y,0]

单次查询时间复杂度为log(n)

例题

模板题

#include<cstdio>
#include<string.h>
#include<algorithm>
using namespace std;
const int MAXN = 4e4+5;
const int MAXM = 2e2+5;

int tot;
int n, m;
int dis[MAXN];
int dep[MAXN];
int head[MAXN];
bool vis[MAXN];
int to[MAXN<<1];
int fa[MAXN][25];
int nxt[MAXN<<1];
int val[MAXN<<1];

void add(int x,int y,int z) {
    nxt[++tot] = head[x];
    head[x] = tot;
    to[tot] = y;
    val[tot] = z;
}

void dfs(int now,int dad) {
    if(vis[now]) return;
    vis[now] = true;
    fa[now][0] = dad;

    for(int i = head[now]; i ;i = nxt[i]) {
        if(to[i] == dad) continue;
        dis[to[i]] = dis[now] + val[i];
        dep[to[i]] = dep[now] + 1;
        dfs(to[i],now);
    }
}

int lca(int x,int y) {
    if(dep[x] > dep[y])
        swap(x,y);
    
    for(int i = 20; i >= 0; i--)
        if((dep[y] - (1<<i) ) >= dep[x])
            y = fa[y][i];
    
    if(x == y) return x;

    for(int i = 20; i >= 0; i--)
        if(fa[x][i] != fa[y][i]) {
            x = fa[x][i];
            y = fa[y][i];
        }
    return fa[x][0];
}

int t;

void inp() {
    for(int i = 1; i < n; i++) {
        int x, y, z;
        scanf("%d %d %d",&x,&y,&z);
        add(x,y,z);
        add(y,x,z);
    }
}

void init() {
    memset(dis,0,sizeof(dis));
    memset(vis,0,sizeof(vis));
    memset(dep,0,sizeof(dep));
    memset(head,0,sizeof(head));
    tot = 0;
}

int main() {

    scanf("%d",&t);
    while(t) {
        scanf("%d %d",&n,&m);
        init();
        inp();

        dfs(1,1);

        for(int i = 1; i <= 20; i++)
            for(int j = 1; j <= n; j++)
                fa[j][i] = fa[fa[j][i-1]][i-1];

        for(int i = 1; i <= m; i++) {
            int x,y;
            scanf("%d %d",&x,&y);
            printf("%d\n",dis[x] + dis[y] - 2*dis[lca(x,y)]);
        }
        t--;
    }

    return 0;
}
