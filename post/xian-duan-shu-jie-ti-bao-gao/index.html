    最大子段和

    题目链接

    #include<cstdio>
#include<algorithm>
using namespace std;

int n,m;
int a[500005];

struct Node{

    int l,r;
    int sum;
    int Max;
    int lm,rm;
}node[2000005];

void pushup(int now){

    node[now].sum = node[now<<1].sum+node[now<<1|1].sum;
    node[now].lm = max(node[now<<1].lm,node[now<<1].sum+node[now<<1|1].lm);
    node[now].rm = max(node[now<<1|1].rm,node[now<<1|1].sum+node[now<<1].rm);
    node[now].Max = max(max(node[now<<1].Max,node[now<<1|1].Max),node[now<<1].rm+node[now<<1|1].lm);
}

void build(int now,int l,int r){

    node[now].l = l;
    node[now].r = r;

    if(l == r)
    {
        node[now].sum = a[l];
        node[now].lm = a[l];
        node[now].rm = a[l];
        node[now].Max = a[l];
        return;
    }
    int mid = (l+r)>>1;

    build(now<<1,l,mid);
    build(now<<1|1,mid+1,r);

    pushup(now);
}

void modify(int now,int to,int k){

    int l = node[now].l;
    int r = node[now].r;

    if(l == r)
    {
        // a[now] = k;
        node[now].sum = k;
        node[now].Max = k;        
        node[now].lm = k;
        node[now].rm = k;

        return;
    }
    int mid = (l+r)>>1;

    if(to <= mid) modify(now<<1,to,k);
    else modify(now<<1|1,to,k);

    pushup(now);
}

Node ask(int now,int L,int R){

    int l = node[now].l;
    int r = node[now].r;

    if(L <= l && R >= r) return node[now];

    int mid = (l+r)>>1;

    if(mid >= R) return ask(now<<1,L,R);
    else if(mid < L) return ask(now<<1|1,L,R);
    else
    {
        Node ans;
        Node left = ask(now<<1,L,R);
        Node right = ask(now<<1|1,L,R);
        ans.sum = left.sum + right.sum;
        ans.lm = max(left.lm,left.sum+right.lm);
        ans.rm = max(right.rm,right.sum+left.rm);
        ans.Max = max(left.rm+right.lm,max(left.Max,right.Max));
        return ans;
    }
}

int main(){

    int k,x,y;

    scanf("%d %d",&n,&m);

    for(int i = 1; i <= n; i++)
        scanf("%d",&a[i]);

    build(1,1,n);
    
    for(int i = 1; i <= m; i++)
    {
        scanf("%d %d %d",&k,&x,&y);
        if(k == 1)
        {
            if(x > y)
            {
                int t = x;
                x = y;
                y = t;
            }
            Node ans = ask(1,x,y);
            printf("%d\n",ans.Max);
        }
        if(k == 2)
            modify(1,x,y);
    }

    return 0;
}

    本题是给一个木板的不同区间染色，一共有30种颜色，可以考虑用一个二进制数字来储存该区间内的数被染了哪一种颜色，线段树中第i个结点存储的数字的二进制第j位为1，则该结点被染成了颜色j，若为0则没有染成颜色j，区间颜色个数统计使用按位与运算，将最终结果进行二进制拆分统计为1的个数即可。

    #include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int MAXN = 1e5+5;

int n,t,o;
int le[MAXN<<2];
int ri[MAXN<<2];
int tag[MAXN<<2];
int color[MAXN<<2];

inline int read() {
    int x = 0;
    char c = getchar();
    while(c < '0' || c > '9')
        c = getchar();
    while(c >= '0' && c <= '9') {
        x = x*10 + c-'0';
        c = getchar();
    }
    return x;
}

int ls(int p) {return p<<1;}
int rs(int p) {return p<<1|1;}
void pushup(int p) {color[p] = color[ls(p)] | color[rs(p)];}

void upd(int p,int v) {
    color[p] = v;
    tag[p] = v;
}

void pushdown(int p) {
    if(tag[p]) {
        upd(ls(p),tag[p]);
        upd(rs(p),tag[p]);
        tag[p] = 0;
    }
}

void build(int now,int l,int r) {
    le[now] = l;
    ri[now] = r;
    color[now] = 0;
    tag[now] = 0;
    if(l == r) {
        color[now] = 1;
        return;
    }
    int mid = (l+r)>>1;
    build(ls(now),l,mid);
    build(rs(now),mid+1,r);
    pushup(now);
}

void modify(int now,int l,int r,int v) {
    int L = le[now];
    int R = ri[now];
    if(r < L || l > R) return;
    if(l <= L && r >= R) {
        tag[now] = 1<<(v-1);
        color[now] = 1<<(v-1);
        return;
    }
    pushdown(now);
    int mid = (L+R)>>1;
    if(l <= mid) modify(ls(now),l,r,v);
    if(r > mid) modify(rs(now),l,r,v);
    pushup(now);
}

int query(int now,int l,int r) {
    int L = le[now];
    int R = ri[now];
    if(r < L || l > R) return 0;
    if(l <= L && r >= R) {
        return color[now];
    }
    int ans = 0;
    int mid = (L+R)>>1;
    pushdown(now);
    if(l <= mid) ans |= query(ls(now),l,r);
    if(r > mid) ans |= query(rs(now),l,r);
    return ans;
}

int main() {
    while(~scanf("%d %d %d",&n,&t,&o)) {
        int x, y, z;
        char c[2];
        build(1,1,n);
        while(o--) {
            scanf("%s",c);
            if(c[0] == 'C') {
                scanf("%d %d %d",&x,&y,&z);
                if(x > y)
                    swap(x,y);
                modify(1,x,y,z);
            }
            else if(c[0] == 'P') {
                scanf("%d %d",&x,&y);
                if(x > y)
                    swap(x,y);
                int cnt = 0;
                int tot = query(1,x,y);
                while(tot) {
                    if(tot&1) cnt++;
                    tot >>= 1;
                }
                printf("%d\n",cnt);
            }
        }
    }

    return 0;
}
